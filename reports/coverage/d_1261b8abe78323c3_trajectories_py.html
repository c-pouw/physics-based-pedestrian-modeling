55def prune_short_trajectories(trajectories: pd.DataFrame, config: DictConfig) -> pd.DataFrame: 

56 """Remove short trajectories that are most likely noise. 

57 

58 The threshold for the minimum trajectory length is set in the configuration object 

59 under the key config.params.minimum_trajectory_length. 

60 

61 Args: 

62 trajectories: The DataFrame containing the trajectories. 

63 config: The configuration object. 

64 

65 Returns: 

66 The DataFrame without short trajectories. 

67 """ 

68 trajectories["traj_len"] = trajectories.groupby(["Pid"])["Pid"].transform("size") 

69 trajectories = trajectories[trajectories.traj_len > config.params.minimum_trajectory_length].copy() 

70 log.info("Short trajectories with less than %s observations removed.", config.params.minimum_trajectory_length) 

71 return trajectories

74def add_trajectory_index(df: pd.DataFrame, config: DictConfig) -> pd.DataFrame: 

75 """Add a column with the observation/step numbers of each trajectory. 

76 

77 Args: 

78 df: The DataFrame to add the trajectory step/observation to. 

79 config: The configuration object. 

80 

81 Returns: 

82 The DataFrame with the trajectory step/observation added. 

83 """ 

84 # pid_col, time_col = params.colnames.Pid, params.colnames.time 

85 pid_col, time_col = "Pid", "time" 

86 df.sort_values(by=[pid_col, time_col], inplace=True) 

87 df["k"] = df.groupby(pid_col)[pid_col].transform(lambda x: np.arange(x.size)) 

88 log.info("Trajectory step added.") 

89 return df

92def compute_velocity_from_positions(df: pd.DataFrame, config: DictConfig) -> pd.DataFrame: 

93 """Add velocity to dataframe with trajectories. 

94 

95 This function calculates the velocity of each pedestrian in the input DataFrame 

96 based on their position data. The velocity is calculated using the Savitzky-Golay 

97 filter with a polynomial order of 1. 

98 

99 Args: 

100 df: The input DataFrame with the trajectories. 

101 

102 Returns: 

103 The trajectories DataFrame with velocity columns added. 

104 """ 

105 uf = config.params.colnames.get("uf", None) 

106 vf = config.params.colnames.get("vf", None) 

107 

108 if uf is not None or vf is not None: 

109 log.warning("Columns for velocity found in parameters, no need to calculate velocity.") 

110 return df 

111 

112 groupby = "Pid" 

113 xpos = "xf" 

114 ypos = "yf" 

115 pos_to_vel = {"xf": "uf", "yf": "vf"} 

116 window_length = config.params.velocity_window_length 

117 # window_length = parameters.minimum_trajectory_length - 1 

118 for direction in [xpos, ypos]: 

119 df.loc[:, pos_to_vel[direction]] = df.groupby([groupby])[direction].transform( 

120 lambda x: savgol_filter(x, window_length=window_length, polyorder=1, deriv=1, mode="interp") * config.params.fps 

121 ) 

122 log.info("Velocities 'uf' and 'vf' added.") 

123 return df
